数学問題計算テンプレ

ホームのやつを最初に置いていること前提です

・素数判定 //O(√N)


bool isprime = true;
for (int i = 2; i*i <= N; i++) {
  if (N%i == 0) isprime = false;
}

・素因数分解 //O(√N)


map<ll,ll> prime_factorization(ll N) {
  map<ll,ll> res;
  for (ll i = 2; i*i <= N; i++) {
    while (N % i == 0) {
      res[i]++;
      N /= i;
    }
  }
  if (N != 1) res[N]++;
  return res;
}

・エラトステネスの篩(N以下の素数を数える) //O(NloglogN)


vector<bool> isprime;
vector<int> prime;
void eratosthenes(ll N) {
  isprime.assign(N+1,true); prime.clear();
  isprime[0] = isprime[1] = false;
  for (ll i = 2; i*i <= N; i++) {
    if (isprime[i]) {
      for (ll j = i*i; j <= N; j += i) isprime[j] = false;
    }
  }
  for (ll i = 2; i <= N; i++) {
    if (isprime[i]) prime.push_back(i);
  }
}

・繰り返し二乗法、逆元の計算 //O(log e)


ll modpow(ll x,ll e) {
  ll r = 1;
  while (e) {
    if (e & 1) r = r*x%MOD;
    x = x*x%MOD;
    e /= 2;
  }
  return r;
}

ll inv(ll x) {
  return modpow(x,MOD-2);
}

・場合の数でよく使うやつまとめ(initで初期化) //O(N)


vector<ll> fact,invfact;
void init(ll N) {
  fact.resize(N+1); invfact.resize(N+1);
  fact[0] = 1;
  for (int i = 1; i <= N; i++) fact[i] = fact[i-1]*i%MOD;
  invfact[N] = inv(fact[N]);
  for (int i = N ; i > 0; i--) invfact[i-1] = invfact[i]*i%MOD;
}

ll nPr(ll n,ll r) {
  if (r > n || r < 0) return 0;
  return fact[n]*invfact[n-r]%MOD;
}

ll nCr(ll n,ll r) {
  if (r > n || r < 0) return 0;
  return fact[n]*invfact[r]%MOD*invfact[n-r]%MOD;
}

ll nHr(ll n,ll r) {
  return nCr(n+r-1,r);
}
